        """ Returns a unit vector in the direction of this vector. """
        length = self.Magnitude()
        if length != 0:
            return self / length
        return Vector2D(self)
    


[docs]    def GetAngle(self):
        """ Returns the angle of this vector. """
        if self.x == self.y == 0:
            return 0
        else:
            return degrees(atan2(self.y, self.x))
