    """A Component to help with positioning of GameObjects and all it's
    attached children with transforms. On changing of position, children will
    re-position themselves relative to their parent's position.
    """
    
    def __init__(self, *position):
        """
        :param position: The initial position for the transform.
        :type position: tuple.
        """
        super(Transform, self).__init__()
        
        if position is ():
            self.__local_position = Vector2D.Zero()
        else:
            self.__local_position = Vector2D(position)

        """Gets the parent of the parent.

    def position(self):
        """Gets the transform position. 
        
        :returns: Vector2D -- The transform position in world space.
        """
        if self.grandparent and hasattr(self.grandparent, 'transform'):
            return self.grandparent.transform.position + self.__local_position
        else:
            return self.local_position
